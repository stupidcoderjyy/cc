%%TOKEN

number: "\d+(\.\d+)?";
id: "\a\w*";

%%

%%SYNTAX

stmt : @id '=' expr
     | expr
     ;

expr : expr '+' $1L expr %1L
     | expr '-' $1L expr %1L
     | expr '*' $2L expr %2L
     | expr '/' $2L expr %2L
     | expr '^' $3L expr %3L
     | '-' expr %5
     | '(' expr ')'
     | @number
     | @id
     ;

%%
