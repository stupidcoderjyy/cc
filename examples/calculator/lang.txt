%%TOKEN

number: "\d+(\.\d+)?";
id: "\a\w*";
constant: "$[A-Z0-9_]+";

%%

%%SYNTAX

stmt : @id '=' expr
     | expr
     ;

expr : expr '+' $1L expr %1L
     | expr '-' $1L expr %1L
     | expr '*' $2L expr %2L
     | expr '/' $2L expr %2L
     | expr '^' $3L expr %3L
     | '-' expr %5
     | '(' expr ')'
     | @id '(' args ')'
     | @number
     | @id
     | @constant
     ;

args : arg
     | args ',' arg
     ;

arg  : expr
     ;

%%
